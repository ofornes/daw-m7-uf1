23  import org.springframework.context.annotation.Configuration;

25  import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
26  import org.springframework.security.core.userdetails.User;

30  import org.springframework.security.crypto.password.PasswordEncoder;
31  import org.springframework.security.provisioning.InMemoryUserDetailsManager;
32  import org.springframework.security.web.SecurityFilterChain;
33  import org.springframework.security.web.util.matcher.AntPathRequestMatcher;

35  import cat.albirar.daw.cocktails.controllers.MainController;

39  /**
40   * Configuració de l'aplicaició.
41   * 
42   * @author Octavi Forn&eacute;s <mailto:ofornes@albirar.cat[]>
43   * @since 1.0.0
44   */
45  @Configuration
46  @ComponentScan(basePackageClasses = { MainController.class, CocktailsServiceImpl.class })
47  @EnableWebSecurity
48  public class CocktailsConfiguration {
49  	private static final String ADMIN_ENC_PASS = "40a12d2b7f546c624461b26bd41573e697e91466dc80ef42acb46685a41b961648422e4529fcbef2fdaf79c7edfbc5e737bed9c224d93ecd26a7f5e028bfa3ed";
50  	private static final String PEPE_ENC_PASS = "c63c4194ea7ab967c7a951a2f784d794318de97710e74bd6d3dcfd680058aecc941973c52e0f74e28aca2840db5a61fb64bfbf974037c34dfb94ebe1b4c860aa";
51  	private static final String MANOLO_ENC_PASS = "3acd3650d01ddf7b2c5fd3488997982684da62c9318a54ee239d5a1f3db72e90b548f10489888ba0da2f0384fa161f319b7d707f2f89cdbe1cc1b6d9ed192fd8";
52  	
53  	@Bean
54  	public UserDetailsService users() {
55  		// The builder will ensure the passwords are encoded before saving in memory
56  		UserBuilder userBuilder = User.builder();
57  		
58  		UserDetails admin = userBuilder
59  				.username("admin")
60  				.password(ADMIN_ENC_PASS)
61  				.roles("USER")
62  				.build();
63  		UserDetails pepe = userBuilder
64  				.username("pepe")
65  				.password(PEPE_ENC_PASS)
66  				.roles("USER")
67  				.build();
68  		UserDetails manolo = userBuilder
69  				.username("manolo")
70  				.password(MANOLO_ENC_PASS)
71  				.roles("USER")
72  				.build();
73  		return new InMemoryUserDetailsManager(admin, pepe, manolo);
74  	}
75  	
76  	@Bean
77  	public PasswordEncoder passwordEncoder() {
78  		return new CocktailsPasswordEncoder();
79  	}
80  	
81  	@Bean
82  	public SecurityFilterChain web(HttpSecurity http) throws Exception {
83  		http
84  			.authorizeHttpRequests(authorize -> authorize
85  					.mvcMatchers("/").permitAll()
86  					.mvcMatchers("/list", "/show/*", "/logout").hasRole("USER")
87  					.anyRequest().denyAll())
88  			.formLogin(form -> form
89  					.loginPage("/")
90  					.defaultSuccessUrl("/", false)
91  					.failureUrl("/")
92  					)
93  			.logout(logout -> logout
94  		            .logoutRequestMatcher(new AntPathRequestMatcher("/logout", "GET"))
95  		            .clearAuthentication(true)
96  		            .invalidateHttpSession(true)
97  		            .logoutSuccessUrl("/")
98  		            )
99  			;
100 		return http.build();
101 	}
102 }
