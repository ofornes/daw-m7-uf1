19  package cat.albirar.daw.cocktails.controllers;
20  
21  import java.util.Optional;
22  
23  import org.springframework.beans.factory.annotation.Autowired;
24  import org.springframework.stereotype.Controller;
25  import org.springframework.ui.Model;
26  import org.springframework.web.bind.annotation.GetMapping;
27  import org.springframework.web.bind.annotation.PathVariable;
28  
29  import cat.albirar.daw.cocktails.models.CocktailDetailBean;
30  import cat.albirar.daw.cocktails.service.ICocktailsService;
31  
32  /**
33   * Main web controller.
34   * @author Octavi Forn&eacute;s <mailto:ofornes@albirar.cat[]>
35   * @since 1.0.0
36   */
37  @Controller
38  public class MainController {
39  	@Autowired
40  	private ICocktailsService cocktailService;
41  	
42  	@GetMapping("/")
43  	public String index() {
44  		return "index";
45  	}
46  	@GetMapping("/list")
47  	public String getList(Model model) {
48  		
49  		model.addAttribute("cocktails", cocktailService.findAll());
50  		return "list";
51  	}
52  	@GetMapping("/show/{id}")
53  	public String getShow(@PathVariable(name = "id", required = true) String id, Model model) {
54  		Optional<CocktailDetailBean> c;
55  
56  		c = cocktailService.findById(id);
57  		if(c.isPresent()) {
58  			model.addAttribute("cocktail", c.get());
59  		}
60  		model.addAttribute("found", c.isPresent());
61  		return "show";
62  	}
63  }
