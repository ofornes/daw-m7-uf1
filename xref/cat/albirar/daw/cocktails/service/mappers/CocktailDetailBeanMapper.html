19  package cat.albirar.daw.cocktails.service.mappers;
20  
21  import java.sql.ResultSet;
22  import java.sql.SQLException;
23  
24  import org.springframework.jdbc.core.RowMapper;
25  import org.springframework.stereotype.Component;
26  
27  import cat.albirar.daw.cocktails.models.CocktailDetailBean;
28  import cat.albirar.daw.cocktails.service.db.ConstantsCocktailsDb;
29  
30  /**
31   * Mapador de JDBC a {@link CocktailDetailBean}.
32   * @author Octavi Forn&eacute;s <mailto:ofornes@albirar.cat[]>
33   * @since 0.1.4
34   */
35  @Component
36  public class CocktailDetailBeanMapper implements RowMapper<CocktailDetailBean> {
37  	/**
38  	 * {@inheritDoc}
39  	 */
40  	@Override
41  	public CocktailDetailBean mapRow(ResultSet rs, int rowNum) throws SQLException {
42  		return CocktailDetailBean.builder()
43  				.id(Integer.toString(rs.getInt(ConstantsCocktailsDb.ID)))
44  				.name(rs.getString(ConstantsCocktailsDb.NAME))
45  				.urlThumb(rs.getString(ConstantsCocktailsDb.THUMB))
46  				.ingredients(rs.getString(ConstantsCocktailsDb.INGREDIENTS))
47  				.instruccions(rs.getString(ConstantsCocktailsDb.INSTRUCTIONS))
48  				.build()
49  				;
50  	}
51  
52  }
