19  package cat.albirar.daw.cocktails.service.impl;
20  
21  import java.io.IOException;
22  import java.util.Collections;
23  import java.util.List;
24  import java.util.Map;
25  import java.util.Optional;
26  import java.util.TreeMap;
27  
28  import javax.annotation.PostConstruct;
29  
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  import org.springframework.beans.factory.annotation.Value;
33  import org.springframework.core.io.Resource;
34  import org.springframework.stereotype.Service;
35  
36  import com.fasterxml.jackson.core.type.TypeReference;
37  import com.fasterxml.jackson.databind.ObjectMapper;
38  
39  import cat.albirar.daw.cocktails.models.CocktailDetailBean;
40  import cat.albirar.daw.cocktails.service.ICocktailsService;
41  
42  /**
43   * Implementació del servei de coctails.
44   * @author Octavi Forn&eacute;s <mailto:ofornes@albirar.cat[]>
45   * @since 1.0.0
46   */
47  @Service
48  public class CocktailsServiceImpl implements ICocktailsService {
49  	private static final Logger LOGGER = LoggerFactory.getLogger(CocktailsServiceImpl.class);
50  	
51  	@Value("classpath:/cocktails.json")
52  	private Resource jsonCocktails;
53  	
54  	private List<CocktailDetailBean> cocktails = null;
55  	private Map<String, CocktailDetailBean> cocktailsMap = null;
56  	
57  	@PostConstruct
58  	private void initList() {
59  		ObjectMapper om;
60  		
61  		om = new ObjectMapper();
62  		try {
63  			cocktails = om.readValue(jsonCocktails.getInputStream(), new TypeReference<List<CocktailDetailBean>>(){});
64  			cocktailsMap = Collections.synchronizedMap(new TreeMap<>());
65  			for(CocktailDetailBean c : cocktails) {
66  				cocktailsMap.put(c.getId(), c);
67  			}
68  		} catch (IOException e) {
69  			LOGGER.error("No s'ha pogut carregar la llista de coctails!", e);
70  			throw new RuntimeException("No s'ha pogut carregar la llista de coctails!", e);
71  		}
72  	}
73  	/**
74  	 * {@inheritDoc}
75  	 */
76  	@Override
77  	public List<CocktailDetailBean> findAll() {
78  		return Collections.unmodifiableList(cocktails);
79  	}
80  
81  	/**
82  	 * {@inheritDoc}
83  	 */
84  	@Override
85  	public Optional<CocktailDetailBean> findById(String id) {
86  		if(cocktailsMap.containsKey(id)) {
87  			return Optional.of(cocktailsMap.get(id));
88  		}
89  		return Optional.empty();
90  	}
91  }
